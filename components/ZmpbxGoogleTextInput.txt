import { Image, View, TextInput, FlatList, TouchableOpacity, Text } from "react-native";
import { useState, useRef } from "react";

import { icons } from "@/constants";
import { GoogleInputProps } from "@/types/type";

const mapboxToken = process.env.EXPO_PUBLIC_MAPBOX_TOKEN;

const GoogleTextInput = ({
  icon,
  initialLocation,
  containerStyle,
  textInputBackgroundColor,
  handlePress,
}: GoogleInputProps) => {
  const [query, setQuery] = useState(initialLocation || "");
  const [places, setPlaces] = useState<any[]>([]);
  const debounceTimeout = useRef<number | null>(null); 

  const handleSearch = (text: string) => {
    setQuery(text);
    if (debounceTimeout.current) clearTimeout(debounceTimeout.current);

    debounceTimeout.current = window.setTimeout(async () => { 
      if (!text) return setPlaces([]);

      try {
        const res = await fetch(
          `https://api.mapbox.com/geocoding/v5/mapbox.places/${encodeURIComponent(
            text
          )}.json?autocomplete=true&limit=5&access_token=${mapboxToken}`
        );
        const data = await res.json();
        setPlaces(data.features || []);
      } catch (error) {
        console.error("Mapbox autocomplete error:", error);
        setPlaces([]);
      }
    }, 200);
  };

  const handleSelectPlace = (place: any) => {
    handlePress({
      latitude: place.center[1],
      longitude: place.center[0],
      address: place.place_name,
    });
    setQuery(place.place_name);
    setPlaces([]);
  };

  return (
    <View
      className={`flex flex-row items-center justify-center relative z-50 rounded-xl ${containerStyle}`}
    >
      <View style={{ flex: 1 }}>
        <TextInput
          value={query}
          placeholder={initialLocation ?? "Where do you want to go?"}
          placeholderTextColor="gray"
          style={{
            backgroundColor: textInputBackgroundColor || "white",
            fontSize: 16,
            fontWeight: "600",
            marginTop: 5,
            width: "100%",
            borderRadius: 200,
            paddingHorizontal: 15,
          }}
          onChangeText={handleSearch}
        />
        {places.length > 0 && (
          <FlatList
            data={places}
            keyExtractor={(item) => item.id}
            renderItem={({ item }) => (
              <TouchableOpacity onPress={() => handleSelectPlace(item)}>
                <Text
                  style={{
                    padding: 10,
                    borderBottomWidth: 1,
                    borderBottomColor: "#eee",
                    backgroundColor: textInputBackgroundColor || "white",
                  }}
                >
                  {item.place_name}
                </Text>
              </TouchableOpacity>
            )}
            style={{
              position: "absolute",
              top: 45,
              width: "100%",
              maxHeight: 200,
              borderRadius: 10,
              zIndex: 99,
              shadowColor: "#d4d4d4",
            }}
          />
        )}
      </View>

      {icon && (
        <View className="justify-center items-center w-6 h-6 ml-2">
          <Image
            source={icon || icons.search}
            className="w-6 h-6"
            resizeMode="contain"
          />
        </View>
      )}
    </View>
  );
};

export default GoogleTextInput;
